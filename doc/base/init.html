页面结构

本节介绍B-JUI框架的初始化方法。初始化方法需要在主页面DOM加载完成后执行，index.html页面有完整的代码

示例代码如下：

                $(function() {
                    BJUI.init({
                        JSPATH       : 'BJUI/',         //[可选]框架路径
                        PLUGINPATH   : 'BJUI/plugins/', //[可选]插件路径
                        loginInfo    : {url:'login_timeout.html', title:'登录', width:400, height:200}, // 会话超时后弹出登录对话框
                        statusCode   : {ok:200, error:300, timeout:301}, //[可选] ajax回调函数的状态码
                        ajaxTimeout  : 5000, //[可选]全局Ajax请求超时时间(毫秒)
                        alertTimeout : 3000, //[可选]信息提示[info/correct]自动关闭延时(毫秒)
                        pageInfo     : {pageCurrent:'pageCurrent', pageSize:'pageSize', orderField:'orderField', orderDirection:'orderDirection'}, //[可选]分页参数key
                        keys         : {statusCode:'statusCode', message:'message'}, //[可选] ajax回调函数的key
                        ui           : {showSlidebar:true, hideMode:'display'}, //[可选]showSlidebar:左侧菜单栏默认显示状态[true|false]；hideMode:navtab组件切换的隐藏方式，支持的值有’display’，’offsets’负数偏移位置的值，默认值为’display’
                        debug        : true,    // [可选]调试模式 [true|false，默认false]
                        theme        : 'purple' // 若有Cookie['bjui_theme'],优先选择Cookie['bjui_theme']。皮肤[六种皮肤:default, orange, purple, blue, red, green]
                    })
                })
